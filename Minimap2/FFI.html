def self.mm_set_opt(preset, io, mo)
  ptr = case preset
        when 0, nil
          ::FFI::Pointer.new(:int, 0)
        else
          ::FFI::MemoryPointer.from_string(preset.to_s)
        end
  mm_set_opt_raw(ptr, io, mo)
end
